25      @Override
26      protected void generateInfo() {
27          StringBuilder result = new StringBuilder();
28          result.append("================================================\n");
29          if (getAlgorithmName() != null) {
30              result.append("Clusterer: " + getAlgorithmName() + "\n");
31          }
32          setInstances(0);
33          for (int i : clusters) {
34              setInstances(getInstances() + i);
35          }
36          result.append("Total instances: " + getInstances() + "\n");
37          result.append("Number of clusters: " + clusters.length + "\n");
38          result.append("Test time (HH:MM:SS.MS): " + timeToString(getTestTime()));
39          result.append("\n------------------------------------------------\n");
40  
41          for (int i = 0; i < clusters.length; i++) {
42              result.append("Cluster " + i + ": " + clusters[i]);
43              double percent = (PERCENT_BASE * clusters[i]) / (double) getInstances();
44              result.append(String.format(" instances (%.2f%%)\n", percent));
45          }
46          result.append("================================================\n");
47          setInfo(result.toString());
48          setModified(false);
49      }
