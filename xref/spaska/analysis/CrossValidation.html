74  	@Override
75  	public ClassifierStatistics analyze(Dataset dataset) {
76  		setData(dataset);
77  
78  		long startTime = System.currentTimeMillis();
79  
80  		// how many elements will have every dataset
81  		int[] datasetsCount = new int[folds];
82  
83  		List<Instance> instances = data.getElements();
84  		int instancesCount = instances.size();
85  
86  		int instancesInGroup = instancesCount / folds;
87  		Arrays.fill(datasetsCount, instancesInGroup);
88  		int cnt = instancesInGroup * folds;
89  		int tmpCnt = 0;
90  		while (cnt++ < instancesCount) {
91  			datasetsCount[tmpCnt++]++;
92  		}
93  
94  		Dataset[] datasets = data.getSubDataSets(datasetsCount, seed);
95  		int classIndex = data.getClassIndex();
96  		ValueType type = data.getAttributes().get(classIndex).getType();
97  		ClassifierStatistics result = new ClassifierStatistics(type);
98  		if (type == ValueType.Nominal) {
99  			result.setClassNames(data.getAllClassNamesArray());
100 		}
101 		result.setAlgorithmName(classifier.getName());
102 
103 		for (int testInd = 0; testInd < folds; testInd++) {
104 			// build classifier for all test datasets
105 			// first merge them
106 			Dataset[] testDatasets = new Dataset[datasets.length - 1];
107 			tmpCnt = 0;
108 			Dataset testDataset = null;
109 			for (int i = 0; i < folds; i++) {
110 				if (i == testInd) {
111 					testDataset = datasets[i];
112 					continue;
113 				}
114 				testDatasets[tmpCnt++] = datasets[i];
115 			}
116 			Dataset trainDataset = Dataset.merge(testDatasets);
117 
118 			classifier.buildClassifier(trainDataset);
119 
120 			// test classifier
121 			for (Instance currentInstance : testDataset.getElements()) {
122 				Value v = classifier.classifyInstance(currentInstance);
123 
124 				switch (v.getType()) {
125 				case Nominal:
126 					result.addNominalInfo(data.getClassName(currentInstance),
127 							((NominalValue) v).getValue());
128 					break;
129 				case Numeric:
130 					result.addNumericInfo((Double) data.getClassValue(
131 							currentInstance).getValue(), (Double) v.getValue());
132 					break;
133 				default:
134 					throw new RuntimeException("other attribute type added");
135 				}
136 			}
137 		}
138 
139 		result.setTestTime(System.currentTimeMillis() - startTime);
140 		result.setAdditionalInfo(String.format(
141 				"Additional info :\n------------------\n%s", classifier.toString()));
142 		return result;
143 	}
