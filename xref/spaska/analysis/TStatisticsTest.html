63  	/**
64  	 * Decides if the Null hypothesis can be rejected according to the preset
65  	 * level of significance (1-alpha)
66  	 * 
67  	 * @param tResult
68  	 *            the value for the T distribution random variable (t statistic)
69  	 * @param df
70  	 *            degrees of freedom
71  	 * @return whether or not to reject H0
72  	 */
73  	public boolean shouldRejectNull(double tResult, int df) {
74  		double quantile = 0;
75  		boolean shouldReject = false;
76  		if (testType == TestType.TwoSided) { // (!=) case
77  			quantile = t.quantile(alpha / 2, df);
78  			if (Math.abs(tResult) > Math.abs(quantile)) {
79  				shouldReject = true;
80  			}
81  		} else {
82  			if (testType == TestType.OneSidedLessThan) {// (<) case
83  				quantile = t.quantile(alpha, df);
84  				if (tResult < quantile) {
85  					shouldReject = true;
86  				}
87  			} else {// (>) case
88  				quantile = t.quantile(1 - alpha, df);
89  				if (tResult > quantile) {
90  					shouldReject = true;
91  				}
92  			}
93  		}
94  		return shouldReject;
95  	}

102 	/**
103 	 * The random variable (mean - a)/(dev/n) has a T distribution. The method
104 	 * calculates the value for this variable given the two sample populations.
105 	 * Since we are using a paired test, the value for a is 0.0
106 	 * 
107 	 * @param fstSample
108 	 *            first sample population
109 	 * @param sndSample
110 	 *            second sample population
111 	 * @return calculated t statistic
112 	 */
113 	public double getT(double[] fstSample, double[] sndSample) {
114 		if (!haveEqualSize(fstSample, sndSample)) {
115 			throw new IllegalArgumentException();
116 		}
117 		double[] differences = getDifferences(fstSample, sndSample);
118 		double mean = EmpiricalTest.getMean(differences);
119 		double dev = EmpiricalTest.getStandardDeviation(differences);
120 		double tResult = ((mean - 0.0) * Math.sqrt(fstSample.length)) / dev;
121 		return tResult;
122 	}

124 	/**
125 	 * Calculates the p-value for the current test. If the p-value is less than
126 	 * alpha, we can reject the null hypothesis
127 	 * 
128 	 * @param fstSample
129 	 *            first random sample
130 	 * @param sndSample
131 	 *            second random sample
132 	 * @return p-value
133 	 */
134 	public double getPValue(double[] fstSample, double[] sndSample) {
135 		double tResult = getT(fstSample, sndSample);
136 		double df = fstSample.length - 1;
137 		double pValue = 0;
138 		if (testType == TestType.TwoSided) { // (!=) case
139 			pValue = t.cumulative(-Math.abs(tResult), df) * 2;
140 		} else {
141 			if (testType == TestType.OneSidedLessThan) {// (<) case
142 				pValue = t.cumulative(tResult, df);
143 			} else {// (>) case
144 				pValue = 1 - t.cumulative(tResult, df);
145 			}
146 		}
147 		return pValue;
148 	}
