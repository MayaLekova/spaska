6    * Compares two evaluation results based on the T test with p-value.

68      private boolean haveEqualSize(double[] fstSample, double[] sndSample) {
69          return fstSample != null && sndSample != null

73      private double[] getDifferences(double[] fstSample, double[] sndSample) {
74          double[] differences = new double[fstSample.length];
75          for (int i = 0; i < fstSample.length; i++) {
76              differences[i] = fstSample[i] - sndSample[i];
77          }
78          return differences;
79      }

81      // Public methods
82      /**
83       * Decides if the Null hypothesis can be rejected according to the preset
84       * level of significance (1-alpha).
85       * 
86       * @param tResult
87       *            the value for the T distribution random variable (t statistic)
88       * @param df
89       *            degrees of freedom
90       * @return whether or not to reject H0
91       */
92      public boolean shouldRejectNull(double tResult, int df) {
93          double quantile = 0;
94          boolean shouldReject = false;
95          if (testType == TestType.TwoSided) { // (!=) case
96              quantile = t.quantile(alpha / 2, df);
97              if (Math.abs(tResult) > Math.abs(quantile)) {
98                  shouldReject = true;
99              }
100         } else {
101             if (testType == TestType.OneSidedLessThan) { // (<) case
102                 quantile = t.quantile(alpha, df);
103                 if (tResult < quantile) {
104                     shouldReject = true;
105                 }
106             } else { // (>) case
107                 quantile = t.quantile(1 - alpha, df);
108                 if (tResult > quantile) {
109                     shouldReject = true;
110                 }
111             }
112         }
113         return shouldReject;
114     }

117     public boolean shouldRejectNull(double[] fstSample, double[] sndSample) {
118         double tResult = getT(fstSample, sndSample);
119         return shouldRejectNull(tResult, fstSample.length - 1);

122     /**
123      * The random variable (mean - a)/(dev/n) has a T distribution. The method
124      * calculates the value for this variable given the two sample populations.
125      * Since we are using a paired test, the value for a is 0.0
126      * 
127      * @param fstSample
128      *            first sample population
129      * @param sndSample
130      *            second sample population
131      * @return calculated t statistic
132      */
133     public double getT(double[] fstSample, double[] sndSample) {
134         if (!haveEqualSize(fstSample, sndSample)) {
135             throw new IllegalArgumentException();
136         }
137         double[] differences = getDifferences(fstSample, sndSample);
138         double mean = EmpiricalTest.getMean(differences);
139         double dev = EmpiricalTest.getStandardDeviation(differences);
140         double tResult = ((mean - 0.0) * Math.sqrt(fstSample.length)) / dev;
141         return tResult;
142     }

144     /**
145      * Calculates the p-value for the current test. If the p-value is less than
146      * alpha, we can reject the null hypothesis
147      * 
148      * @param fstSample
149      *            first random sample
150      * @param sndSample
151      *            second random sample
152      * @return p-value
153      */
154     public double getPValue(double[] fstSample, double[] sndSample) {
155         double tResult = getT(fstSample, sndSample);
156         double df = fstSample.length - 1;
157         double pValue = 0;
158         if (testType == TestType.TwoSided) { // (!=) case
159             pValue = t.cumulative(-Math.abs(tResult), df) * 2;
160         } else {
161             if (testType == TestType.OneSidedLessThan) { // (<) case
162                 pValue = t.cumulative(tResult, df);
163             } else { // (>) case
164                 pValue = 1 - t.cumulative(tResult, df);
165             }
166         }
167         return pValue;
168     }
