60  	public boolean validate() {
61  		List<Attribute> attributes = dataset.getAttributes();
62  		List<Instance> instances = dataset.getElements();
63  		int attrIndex = 0;
64  		for (Attribute attribute : attributes) {
65  			if (attribute.getType().equals(ValueType.Numeric)) {
66  				normalize(attrIndex, instances);
67  			}
68  			attrIndex++;
69  		}
70  
71  		return true;
72  	}
73  
74  	public void normalize(int attrIndex, List<Instance> instances) {
75  		Double maxValue = new Double(0);
76  		Double minValue = new Double(0);
77  		for (Instance element : instances) {
78  			Double value = ((NumericValue) element.getVector().get(attrIndex)).getValue();
79  			if (maxValue < value) {
80  				maxValue = new Double(value);
81  			}
82  			if (minValue > value) {
83  				minValue = new Double(value);
84  			}
85  		}
86  		for (Instance element : instances) {
87  			Double currentValue = ((NumericValue) element.getVector().get(attrIndex)).getValue();
88  			Double normalizedValue = downLimit + (currentValue - minValue) * (upperLimit - downLimit) / (maxValue - minValue);
89  			element.getVector().set(attrIndex, new NumericValue(normalizedValue));
90  
91  		}
92  
93  	}

98  		}
