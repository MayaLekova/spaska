18       * Creates a new value from the given object value.
19       * 
20       * @param val
21       *            the value from which to construct a Value object
22       * @return the newly constructed value object
23       */
24      public static Value createValue(Object val) {
25          if (val instanceof String) {
26              return new NominalValue((String) val);
27          } else if (val instanceof Double) {
28              return new NumericValue((Double) val);
29          }
30          return UnknownValue.getInstance();
31      }

33      /**
34       * Create a new instance from bulk data.
35       * 
36       * @param vector
37       *            instance data
38       * @return the newly created instance
39       */
40      public static Instance createElement(Object[] vector) {
41          List<Value> values = new Vector<Value>();
42          for (Object value : vector) {
43              values.add(createValue(value));
44          }
45          return new Instance(values);
46      }

48      /**
49       * Convert string values to data {@link Value} objects.
50       * 
51       * @param strValues
52       *            the string values to be converted
53       * @param dataset
54       *            the dataset to get the metadata for attributes. This is used
55       *            to help us in the convertion
56       * @return a list of data {@link Value} objects
57       */
58      public static List<Value> createElementData(String[] strValues,
59              Dataset dataset) {
60          List<Value> element = new Vector<Value>();
61          List<Attribute> attributes = dataset.getAttributes();
62          int i = 0;
63          for (Attribute a : attributes) {
64              if (i < strValues.length) {
65                  if (a.getType().equals(ValueType.Numeric)) {
66                      try {
67                          Double val = Double.parseDouble(strValues[i]);
68                          element.add(new NumericValue(val));
69                      } catch (java.lang.NumberFormatException e) {
70                          element.add(UnknownValue.getInstance());
71                      }
72                  } else {
73                      if ((strValues[i].trim()).equals("?")) {
74                          element.add(UnknownValue.getInstance());
75                      } else {
76                          element.add(new NominalValue(strValues[i].trim()));
77                      }
78                  }
79              } else {
80                  element.add(UnknownValue.getInstance());
81              }
82              i++;
83          }
84          return element;
85      }
