114 	private <T> void setEngineArgs(Map<Class<? extends T>, Map<String, String>> classToParameters) throws Exception {
115 		for (Entry<Class<? extends T>, Map<String, String>> entry : classToParameters.entrySet()) {
116 			Class<? extends T> cls = entry.getKey();
117 			Map<String, String> params = (entry.getValue() != null) ? entry.getValue() : Utils.getParamsOfClass(cls);
118 
119 			System.out.println("Set " + cls + " with " + params);
120 			if (Validator.class.isAssignableFrom(cls)) {
121 				getEngine().addValidator((Validator) cls.newInstance(), params);
122 			}
123 			else if (IClusterer.class.isAssignableFrom(cls)) {
124 				getEngine().setClusterer((IClusterer) cls.newInstance(), params);
125 			}
126 		}		
127 	}
