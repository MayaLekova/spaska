75      public void showStatistics(Statistics st) {
76          if (st != null) {
77              if (!getStatisticsDialog().isVisible()) {
78                  getStatisticsDialog().setVisible(true);
79              }
80              area.setText(st.toString());
81          } else {
82              area.setText("No generated statistics yet.");

159     private JDialog getAboutDialog() {
160         if (aboutDialog == null) {
161             aboutDialog = new JDialog(this);
162             aboutDialog.setLayout(new BorderLayout(10, 10));
163             setResizable(false);
164 
165             Font bodyFont = getFont().deriveFont(10f);
166 
167             JPanel panel = new JPanel(new GridBagLayout());
168             GridBagConstraints c = new GridBagConstraints();
169             c.insets = new Insets(2, 2, 2, 2);
170 
171             List<JLabel> labels = new LinkedList<JLabel>();
172             labels.add(new JLabel(""));
173             labels.add(new JLabel(""));
174             for (String s : bundle.get(bundle.APP_NAME).split("\n")) {
175                 JLabel l = new JLabel(s);
176                 l.setFont(bodyFont);
177                 labels.add(l);
178             }
179             c.gridx = 0;
180             c.gridy = 0;
181             c.gridheight = labels.size() + 1;
182             c.anchor = GridBagConstraints.NORTHWEST;
183             panel.add(new JLabel(bundle.getIcon(bundle.get(bundle.APP_ICON))),
184                     c);
185             c.gridx = 1;
186             c.gridy = 0;
187             c.gridheight = 1;
188             c.anchor = GridBagConstraints.NORTHEAST;
189             panel.add(
190                     new JLabel(bundle.get(bundle.APP_TITLE) + " "
191                             + bundle.get(bundle.APP_VERSION)), c);
192 
193             for (JLabel l : labels) {
194                 c.gridy++;
195                 panel.add(l, c);
196             }
197             labels.clear();
198             c.gridx = 0;
199             c.gridy++;
200             c.gridwidth = 2;
201             c.anchor = GridBagConstraints.CENTER;
202 
203             labels.add(new JLabel(""));
204             labels.add(new JLabel(""));
205             labels.add(new JLabel("Credits:"));
206 
207             for (String s : bundle.get(bundle.APP_CERDITS).split("\n")) {

236     @SuppressWarnings(justification = "This is a proper action when exit is called", value = "DM_EXIT")

238     public void actionPerformed(ActionEvent e) {
239         if (e.getActionCommand().equals(Utils.EXIT)) {
240             System.exit(0);
241         } else if (e.getActionCommand().equals(Utils.OPEN_STATISTICS)) {
242             showStatistics(getSelectedTab().getStatistics());
243         } else if (e.getActionCommand().equals(Utils.FILE_DATASET)) {
244             getSelectedTab().openFile();
