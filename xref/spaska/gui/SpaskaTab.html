60      protected <T> void setEngineArgs(
61              Map<Class<? extends T>, Map<String, String>> classToParameters,
62              String context) throws Exception {
63          for (Entry<Class<? extends T>, Map<String, String>> entry : classToParameters
64                  .entrySet()) {
65              Class<? extends T> cls = entry.getKey();
66              Map<String, String> params = (entry.getValue() != null) ? entry
67                      .getValue() : Utils.getParamsOfClass(cls);
68  
69              LOG.info("Set " + cls + " with " + params);
70              if (Validator.class.isAssignableFrom(cls)) {
71                  getEngine().addValidator((Validator) cls.newInstance(), params);
72              } else {
73                  setComponent(cls, params, context);
74              }
75          }
76      }

92      public void openFile() {
93          if (fileChooser == null) {
94              fileChooser = new JFileChooser();
95          }
96          if (openedFile != null) {
97              fileChooser.setCurrentDirectory(openedFile.getParentFile());
98          }
99          fileChooser.setCurrentDirectory(new File("."));
100         if (fileChooser.showOpenDialog(this) == JFileChooser.APPROVE_OPTION) {
101             openedFile = fileChooser.getSelectedFile();
102             propertyChange(Utils.FILE_DATASET, openedFile);
103         }
104     }

118     protected void start() throws Exception {
119         stop();
120         if (thread == null) {
121             thread = new Thread() {
122                 @Override
123                 public void run() {
124                     runAlgorithm();
125                 }
126             };
127             engine.check();
128             thread.setDaemon(true);
129             thread.start();
130         }
131     }

133     protected void stop() {
134         if (thread != null) {
135             thread.interrupt();
136             thread = null;
137         }
138     }

140     protected void showError(Exception e) {
141         stop();
142         setButtonStart();
143         if (e instanceof InputException) {
144             JOptionPane.showMessageDialog(this, e.getMessage(), "Missing Data",
145                     JOptionPane.ERROR_MESSAGE);
146         } else {
147             LOG.error("Error occured in the table", e);
148             JOptionPane.showMessageDialog(this, e, "Algorithm error",
149                     JOptionPane.ERROR_MESSAGE);
150         }
151     }

153     protected void runAlgorithm() {
154         try {

156             MainApp.getInstance().showStatistics(getStatistics());
157         } catch (Exception e) {
158             showError(e);
