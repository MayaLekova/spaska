130 	private <T> void setEngineArgs(Map<Class<? extends T>, Map<String, String>> classToParameters, boolean first) throws Exception {
131 		for (Entry<Class<? extends T>, Map<String, String>> entry : classToParameters.entrySet()) {
132 			Class<? extends T> cls = entry.getKey();
133 			Map<String, String> params = (entry.getValue() != null) ? entry.getValue() : Utils.getParamsOfClass(cls);
134 
135 			System.out.println("Set " + cls + " with " + params);
136 			if (Validator.class.isAssignableFrom(cls)) {
137 				getEngine().addValidator((Validator) cls.newInstance(), params);
138 			}
139 			else if (IClassifier.class.isAssignableFrom(cls)) {
140 				if (first) {
141 					getEngine().setClassifier1((IClassifier) cls.newInstance(), params);
142 				}
143 				else {
144 					getEngine().setClassifier2((IClassifier) cls.newInstance(), params);
145 				}
146 			}
147 			else if (ICompareAnalyzer.class.isAssignableFrom(cls)) {
148 				getEngine().setAnalyzer((ICompareAnalyzer) cls.newInstance(), params);
149 			}
150 		}		
151 	}
