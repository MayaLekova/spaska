23  public final class PackageClassesDiscovery {

25      private static final int CLASS_SUFFIX_LENGTH = 6;
26  
27      private static final Logger LOG = LoggerFactory
28              .getLogger(PackageClassesDiscovery.class);

30      private String packageName;
31  
32      /**
33       * Constructs a package discovery object that can be used to discover
34       * classes in the given package name.
35       * 
36       * @param packageName
37       *            the package name in which to discover classes recursively
38       */
39      public PackageClassesDiscovery(String packageName) {
40          this.packageName = packageName;
41      }
42  
43      /**
44       * Finds subclasses of the given class in the package.
45       * 
46       * @param baseClass
47       *            the class that will be used for filtering it's subclasses when
48       *            searching
49       * @return all subclasses of the given baseClass param in the given package
50       * @param <T>
51       *            the type of classes that should be found
52       */
53      public <T> List<Class<? extends T>> findSubclassesOf(Class<T> baseClass) {
54          ClassLoader classLoader = Thread.currentThread()
55                  .getContextClassLoader();
56          assert classLoader != null;
57          String path = packageName.replace('.', '/');
58          Enumeration<URL> resources;
59          ArrayList<Class<? extends T>> classes = new ArrayList<Class<? extends T>>();
60          try {
61              resources = classLoader.getResources(path);
62              List<File> dirs = new ArrayList<File>();
63              while (resources.hasMoreElements()) {
64                  URL resource = resources.nextElement();
65                  try {
66                      dirs.add(new File(resource.toURI()));
67                  } catch (URISyntaxException e) {
68                      LOG.error(
69                              "Error while converting the classpath directory to uri!",
70                              e);
71                  }
72              }
73              for (File directory : dirs) {
74                  classes.addAll(PackageClassesDiscovery.findClasses(directory,
75                          packageName, baseClass));
76              }
77          } catch (IOException e) {
78              LOG.error("Error while getting classpath resources", e);
79          }
80  
81          return classes;
82      }
83  
84      private static String getClassNameFromFile(File file) {
85          return file.getName().substring(0,
86                  file.getName().length() - CLASS_SUFFIX_LENGTH);
87      }
88  
89      /**
90       * Recursive method used to find all classes in a given directory and
91       * subdirs.
92       * 
93       * @param directory
94       *            The base directory
95       * @param packageName
96       *            The package name for classes found inside the base directory
97       * @return The classes
98       * @throws ClassNotFoundException
99       */
100     private static <T> List<Class<? extends T>> findClasses(File directory,
101             String packageName, Class<? extends T> filteredClasses) {
102         List<Class<? extends T>> classes = new ArrayList<Class<? extends T>>();
103         if (!directory.exists()) {
104             return classes;
105         }
106         File[] files = directory.listFiles();
107         for (File file : files) {
108             if (file.isDirectory()) {
109                 assert !file.getName().contains(".");
110                 classes.addAll(findClasses(file,
111                         packageName + "." + file.getName(), filteredClasses));
112             } else if (file.getName().endsWith(".class")) {
113                 Class<?> clazz;
114                 String className = packageName + '.'
115                         + getClassNameFromFile(file);
116                 try {
117                     clazz = Class.forName(className);
118                 } catch (ClassNotFoundException e) {
119                     LOG.error("Couldn't find the class with name {}",
120                             className, e);
121                     continue;
122                 }
123                 if (clazz == filteredClasses) {
124                     continue;
125                 }
126                 if (filteredClasses.isAssignableFrom(clazz)) {
127                     @SuppressWarnings("unchecked")
128                     Class<? extends T> result = (Class<? extends T>) clazz;
129                     classes.add(result);
130                 }
131             }
132         }
133         return classes;
134     }
