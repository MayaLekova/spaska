36      /**
37       * Constructs a package discovery object that can be used to discover
38       * classes in the given package name.
39       * 
40       * @param packageName
41       *            the package name in which to discover classes recursively
42       */
43      public PackageClassesDiscovery(String packageName) {
44          this.packageName = packageName;
45          this.packagePath = packageName.replace('.', '/');
46      }

48      /**
49       * Finds subclasses of the given class in the package.
50       * 
51       * @param baseClass
52       *            the class that will be used for filtering it's subclasses when
53       *            searching
54       * @return all subclasses of the given baseClass param in the given package
55       * @param <T>
56       *            the type of classes that should be found
57       */
58      public <T> List<Class<? extends T>> findSubclassesOf(Class<T> baseClass) {
59          ClassLoader classLoader = Thread.currentThread()
60                  .getContextClassLoader();
61          assert classLoader != null;
62  
63          Enumeration<URL> resources;
64          List<Class<? extends T>> classes = new ArrayList<Class<? extends T>>();
65          try {
66              resources = classLoader.getResources(packagePath);
67              while (resources.hasMoreElements()) {
68                  URL resource = resources.nextElement();
69                  try {
70                      String dirPath = resource.toURI().toASCIIString()
71                              .replaceFirst("file:", "");
72                      if (dirPath.startsWith("jar:")) {
73                          findClassesInJar(dirPath, classes, baseClass);
74                      } else {
75                          classes.addAll(PackageClassesDiscovery.findClasses(
76                                  new File(dirPath), packageName, baseClass));
77                      }
78                  } catch (URISyntaxException e) {
79                      LOG.error(
80                              "Error while converting the classpath directory to uri!",
81                              e);
82                  }
83              }
84          } catch (IOException e) {
85              LOG.error("Error while getting classpath resources", e);
86          }
87  
88          return classes;
89      }

91      /**
92       * Finds classes in a jar file.
93       * 
94       * @param dirPath
95       *            the dir path
96       * @param classes
97       *            the output classes that is expected
98       * @param baseClass
99       *            the base class that we want to get the subclasses of
100      * @throws IOException
101      *             if any exception occurs while reading the file in the zip
102      * @throws ZipException
103      *             if any exception occurs while reading the zip file
104      */
105     private <T> void findClassesInJar(String dirPath,
106             List<Class<? extends T>> classes, Class<? extends T> baseClass)
107             throws IOException {
108         dirPath = dirPath.replaceFirst("jar:", "").replaceFirst("file:", "")
109                 .replaceAll("!.*$", "");
110         ZipFile file = new ZipFile(new File(dirPath));
111         Enumeration<? extends ZipEntry> entries = file.entries();
112         while (entries.hasMoreElements()) {
113             ZipEntry entry = entries.nextElement();
114             String entryName = entry.getName();
115             if (entryName.startsWith(packagePath)
116                     && entryName.endsWith(".class")) {
117                 String classNameFromFile = getClassNameFromFile(entryName);
118                 addClass(classNameFromFile.replaceAll("/", "."), baseClass,
119                         classes);
120             }
121         }
122     }
123 
124     private static String getClassNameFromFile(String fileName) {
125         return fileName.substring(0, fileName.length() - CLASS_SUFFIX_LENGTH);
126     }
127 
128     /**
129      * Recursive method used to find all classes in a given directory and
130      * subdirs.
131      * 
132      * @param directory
133      *            The base directory
134      * @param packageName
135      *            The package name for classes found inside the base directory
136      * @return The classes
137      * @throws ClassNotFoundException
138      */
139     private static <T> List<Class<? extends T>> findClasses(File directory,
140             String packageName, Class<? extends T> filteredClasses) {
141         List<Class<? extends T>> classes = new ArrayList<Class<? extends T>>();
142         if (!directory.exists()) {
143             return classes;
144         }
145         File[] files = directory.listFiles();
146         for (File file : files) {
147             if (file.isDirectory()) {
148                 assert !file.getName().contains(".");
149                 classes.addAll(findClasses(file,
150                         packageName + "." + file.getName(), filteredClasses));
151             } else if (file.getName().endsWith(".class")) {
152                 String className = packageName + '.'
153                         + getClassNameFromFile(file.getName());
154                 addClass(className, filteredClasses, classes);
155             }
156         }
157         return classes;
158     }

160     /**
161      * Add a single class to the list of found classes.
162      * 
163      * @param className
164      *            the class name to be considered for adding
165      * @param filteredClasses
166      *            the base class we are searching the subclasses of.
167      * @param classes
168      *            the list that needs to be populated with the class if needed
169      */
170     private static <T> void addClass(String className,
171             Class<?> filteredClasses, List<Class<? extends T>> classes) {
172         Class<?> clazz = null;
173         try {
174             clazz = Class.forName(className);
175         } catch (ClassNotFoundException e) {
176             LOG.error("Couldn't find the class with name {}", className, e);
177             return;
178         }
179         if (clazz == filteredClasses) {
180             return;
181         }
182         if (filteredClasses.isAssignableFrom(clazz)) {
183             @SuppressWarnings("unchecked")
184             Class<? extends T> result = (Class<? extends T>) clazz;
185             classes.add(result);
186         }
187     }
