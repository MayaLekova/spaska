21  /**
22   * A dicision tree algorithm based on ID3 and C4.5. Attributes are considered
23   * for splitting according to their gain ratio. The higher the gain ratio, the
24   * bigger the chance the attribute will be chosen for splitting. Numeric
25   * attributes define binary split points of the form (<= value ; > value).
26   * Instances with unknown attributes are copied and then spread among all
27   * possible values of the splitting attribute. These instances are given weight
28   * according to the popularity of the attribute value, however, at this stage
29   * this weight is not considered. New instances which encounter a test, such
30   * that their value for that attribute is unknown, are classified as the most
31   * popular class for that branch.
32   */
33

51      // check if all instances are of class defaultClass
52      private boolean allHaveTheSameClass(List<Instance> instances,
53              Value defaultClass) {
54          for (Instance instance : instances) {
55              if (!defaultClass.equals(datasetService.getClass(instance))) {
56                  return false;
57              }
58          }
59          return true;
60      }

62      // get index of best nominal attribute (according to information theory)
63      private int getBestNominal(NominalInfoService service, boolean[] used) {
64          int[] nominalIndices = datasetService.getNominalIndices();
65          double max = 0, currentRatio;
66          int index = -1;
67          for (int i = 0; i < nominalIndices.length; i++) {
68              if (!used[nominalIndices[i]]) {
69                  currentRatio = service.getGainRatio(nominalIndices[i]);
70                  if (max < currentRatio) {
71                      max = currentRatio;
72                      index = nominalIndices[i];
73                  }
74              }
75          }
76          return index;
77      }

79      // get children of the current node
80      private List<Condition> getChildrenConditions(List<Instance> list,
81              boolean[] used) {
82          List<Condition> children = new ArrayList<Condition>();
83          NominalInfoService infoService = new NominalInfoService(datasetService,
84                  list, used);
85          ContinuousValueService continuousService = Trees.getBestNumeric(list,
86                  infoService.getClassesEntropy(), datasetService);
87          if (infoService.isEmpty() && continuousService.isEmpty()) {
88              return children;
89          }
90          Value majorityClass = infoService.getSiblingsMajorityClass();
91          int nominalIndex = getBestNominal(infoService, used);
92          if (nominalIndex < 0 && continuousService.isEmpty()) {
93              return children;
94          }
95          if (continuousService.isEmpty()
96                  || infoService.getGainRatio(nominalIndex) >= continuousService
97                          .getGainRatio()) {
98              children = Trees.getNominalConditions(nominalIndex, majorityClass,
99                      datasetService);
100             used[nominalIndex] = true;
101         } else {
102             children = Trees.getNumericConditions(
103                     continuousService.getAttributeIndex(),
104                     continuousService.getSplitValue(), majorityClass,
105                     datasetService);
106         }
107         return children;
108     }

110     private void buildTree(Node node, List<Instance> treeInstances,
111             boolean[] used) {
112         // stopping criteria
113         // all - the same class
114         Value firstClass = datasetService.getClass(treeInstances.get(0));
115         if (allHaveTheSameClass(treeInstances, firstClass)) {
116             Condition c = getNodeCondition(node);
117             c.setEffect(firstClass);
118             c.setNumber(treeInstances.size());
119             return;
120         }
121         if (treeInstances.isEmpty()) {
122             // majority vote - parent node
123             Condition c = getNodeCondition(node);
124             c.setEffect(c.getMajorityClass());
125             return;
126         }
127         // zero gain - all tests
128 
129         // for each attribute (numeric or non-used nominal) calculate gain ratio
130         // choose the best attribute
131         List<Condition> children = getChildrenConditions(treeInstances, used);
132         if (children.isEmpty()) { // no patterns extracted from data => possibly
133             // unknown values
134             Condition c = getNodeCondition(node);
135             c.setEffect(c.getMajorityClass());
136             c.setNumber(treeInstances.size());
137             return;
138         }
139         // split on that attribute
140         List<List<Instance>> distribution = Trees.distribute(treeInstances,
141                 children, datasetService);
142         int counter = 0;
143         for (Condition c : children) {
144             Node child = new Node(c);
145             node.addChild(child);
146             boolean[] newUsed = new boolean[used.length];
147             System.arraycopy(used, 0, newUsed, 0, used.length);
148             buildTree(child, distribution.get(counter++), newUsed);
149         }
150     }

152     private Node buildTree(List<Instance> instances) {
153         Node root = new Node(new Object());
154         boolean[] used = new boolean[datasetService.numberOfAttributes()];
155         used[datasetService.classIndex()] = true;
156         buildTree(root, instances, used);
157         return root;
158     }

171     // recursive classification
172     private Value classify(Instance instance, Node node) {
173         if (isLeaf(node)) {
174             return getNodeCondition(node).getEffect();
175         } else {
176             Value result = null;
177             for (Node current : node.getChildren()) {
178                 if (Thread.interrupted()) {
179                     return UnknownValue.getInstance();
180                 }
181                 Condition c = (Condition) current.getValue();
182                 int aIndex = datasetService.getAttributeIndex(c.getAttribute());
183                 Value aValue = instance.getVector().get(aIndex);
184                 if (aValue.getType() == ValueType.Unknown) {
185                     return c.getMajorityClass();
186                 }
187                 if (c.ifTrue(aValue)) {
188                     result = classify(instance, current);
189                     break;
190                 }
191             }
192             return result;
193         }
194     }

202     private boolean isLeaf(Node node) {
203         if (!(node.getValue() instanceof Condition)) {
