60      /* A fixed size queue for storing distances and the corresponding instances */
61      private class DistanceQueue {
62  
63          Pair[] distances; // the queue
64  
65          // construct the queue given the classifier parameter K
66          DistanceQueue(int k) {
67              distances = new Pair[k];
68          }
69  
70          // set default values
71          void clear() {
72              for (int i = 0; i < distances.length; ++i) {
73                  distances[i] = new Pair();
74              }
75          }
76  
77          /*
78           * insert a pair it its proper place in the queue keeping the queue
79           * sorted
80           */
81          void push(double distance, Instance instance) {
82              if (distance >= distances[0].distance) {
83                  return; // do not insert if value is worse than the worst in
84                  // queue
85              }
86  
87              int index = 0;
88              // find the correct index of the new pair
89              while (index < distances.length && distance < distances[index].distance) {
90                  ++index;
91              }
92              --index;
93              // shift previous pairs and insert
94              for (int i = 0; i < index; ++i) {
95                  distances[i] = distances[i + 1];
96              }
97              distances[index] = new Pair(distance, instance);
98          }
99  
100         /*
101          * returns the most frequent class value (if discrete) or mean of class
102          * values (if numeric) in the queue
103          */
104         Value voteForClass() {
105 
106             ValueType type = originalTrainSet.getAttributes().get(classIndex).getType();
107             // 1st case: numeric class value
108             if (type == ValueType.Numeric) {
109                 double sum = 0.0, sumWeights = 0.0;
110                 for (int i = 0; i < distances.length; ++i) {
111                     if (distances[i].distance == 0.0) {
112                         return distances[i].instance.getVector().get(classIndex);
113                     }
114                     double weight = weighted ? (1 / distances[i].distance)
115                             : 1.0;
116                     sumWeights += weight;
117                     sum += weight * (Double) (distances[i].instance.getVector().get(
118                             classIndex).getValue());
119                 }
120                 return new NumericValue(sum / sumWeights);
121             } // 2nd case: nominal class value
122             else {
123                 // a map for storing frequencies/weights
124                 Map<Value, Double> frequencies = new HashMap<Value, Double>();
125                 for (int i = 0; i < distances.length; ++i) {
126                     if (distances[i].distance == 0.0) {
127                         return distances[i].instance.getVector().get(classIndex);
128                     }
129                     Value currentClass = distances[i].instance.getVector().get(
130                             classIndex);
131                     Double oldValue = frequencies.get(currentClass);
132                     double weight = weighted ? (1 / distances[i].distance)
133                             : 1.0; // set weight or frequency
134                     if (weight == Double.NaN) {
135                         weight = 1.0;
136                     }
137                     if (oldValue == null) {
138                         frequencies.put(currentClass, weight);
139                     } else {
140                         frequencies.put(currentClass, oldValue + weight);
141                     }
142                 }
143                 // traverse the map to find the most frequent class value
144                 Iterator<Value> it = frequencies.keySet().iterator();
145                 double maxFrequency = Double.MIN_VALUE;
146                 Value result = null;
147                 while (it.hasNext()) {
148                     Value currentClass = it.next();
149                     Double currentFrequency = frequencies.get(currentClass);
150                     /*
151                      * System.out.print(currentClass + " -> ");
152                      * System.out.println(currentFrequency);
153                      */
154                     if (currentFrequency > maxFrequency) {
155                         maxFrequency = currentFrequency;
156                         result = currentClass;
157                     }
158                 }
159                 //System.out.println("returned class: " + result);
160                 return result;
161             }
162         }

191     /*
192      * fill the queue with distances from the query instance to its K nearest
193      * neighbors
194      */
195     private void calculateDistances(Instance query) {
196 
197         double result = 0;
198         List<Value> queryVector = query.getVector();
199         int vectorSize = originalTrainSet.getAttributes().size();
200         // for optimization
201         queryVectorArray = new Value[queryVector.size()];
202         for (int i = 0; i < queryVectorArray.length; i++) {
203             queryVectorArray[i] = queryVector.get(i);
204         }
205 
206         for (Instance currentInstance : trainSet) {
207 
208             List<Value> currentVector = currentInstance.getVector();
209             result = 0;
210             for (int i = 0; i < vectorSize; ++i) {
211                 if (i == classIndex) {
212                     // skip distance beteween class values
213                     continue;
214                 }
215                 Value currentValue = currentVector.get(i);
216                 result += distanceBetweenValues(queryVectorArray[i],
217                         currentValue, attrTypes[i]);
218             }
219 
220             result = Math.sqrt(result); // distance is Euclidean
221             distanceQueue.push(result, currentInstance);
222         }
223     }

225     /* returns the distance between two values, given their type */
226     private double distanceBetweenValues(Value queryValue, Value currentValue,
227             ValueType attrType) {
228         // 1st case: nominal values
229         if (attrType == ValueType.Nominal) {
230 
231             if (queryValue.getType() != ValueType.Unknown && currentValue.getType() != ValueType.Unknown) {
232                 if (!queryValue.equals(currentValue)) {
233                     return 1;
234                 }
235             } else if (queryValue.getType() == ValueType.Unknown && currentValue.getType() == ValueType.Unknown) {
236                 return 0;
237             } else {
238                 return 1;
239             }
240         } // 2nd case: numeric values
241         else if (attrType == ValueType.Numeric) {
242             if (queryValue.getType() != ValueType.Unknown && currentValue.getType() != ValueType.Unknown) {
243                 double dst = (Double) queryValue.getValue() - (Double) currentValue.getValue();
244                 return dst * dst;
245             } else if ((queryValue.getType() == ValueType.Unknown && currentValue.getType() == ValueType.Unknown)) {
246                 return 0;
247             } else {
248                 return 1;
249             }
250         }
251         return 0;
252     }

254     /* train the knn classifier */
255     public void buildClassifier(Dataset instances) {
256 
257         originalTrainSet = instances;
258         classIndex = instances.getClassIndex();
259         trainSet = new ArrayList<Instance>(instances.getElements().size());
260         List<Instance> givenInstances = instances.getElements();
261         Iterator<Instance> it = givenInstances.iterator();
262         while (it.hasNext()) { // prune instances with missing class value
263             Instance current = it.next();
264             if (current.getVector().get(classIndex).getType() != ValueType.Unknown) {
265                 trainSet.add(current);
266             }
267         }
268         // set attribute types array
269         attrTypes = new ValueType[instances.getAttributesCount()];
270         for (int i = 0; i < attrTypes.length; i++) {
271             attrTypes[i] = instances.getAttributes().get(i).getType();
272         }
273         
274         if(k > trainSet.size()) {
275             setK(trainSet.size());
276             System.err.println("KNN: k >= all neighbors! Setting k to number of neighbors.");
277         }
278     }
