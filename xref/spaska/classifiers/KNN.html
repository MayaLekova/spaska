67          private Instance instance;

79      /**
80       * A fixed size queue for storing distances and the corresponding instances.
81       */
82      private class DistanceQueue {
83  
84          private Pair[] distances; // the queue
85  
86          // construct the queue given the classifier parameter K
87          DistanceQueue(int k) {
88              distances = new Pair[k];
89          }
90  
91          // set default values
92          void clear() {
93              for (int i = 0; i < distances.length; ++i) {
94                  distances[i] = new Pair();
95              }
96          }
97  
98          /*
99           * insert a pair it its proper place in the queue keeping the queue
100          * sorted
101          */
102         void push(double distance, Instance instance) {
103             if (distance >= distances[0].distance) {
104                 return; // do not insert if value is worse than the worst in
105                 // queue
106             }
107 
108             int index = 0;
109             // find the correct index of the new pair
110             while (index < distances.length
111                     && distance < distances[index].distance) {
112                 ++index;
113             }
114             --index;
115             // shift previous pairs and insert
116             for (int i = 0; i < index; ++i) {
117                 distances[i] = distances[i + 1];
118             }
119             distances[index] = new Pair(distance, instance);
120         }
121 
122         /*
123          * returns the most frequent class value (if discrete) or mean of class
124          * values (if numeric) in the queue
125          */
126         Value voteForClass() {
127 
128             ValueType type = originalTrainSet.getAttributes().get(classIndex)
129                     .getType();
130             // 1st case: numeric class value
131             if (type == ValueType.Numeric) {
132                 double sum = 0.0, sumWeights = 0.0;
133                 for (int i = 0; i < distances.length; ++i) {
134                     if (distances[i].distance == 0.0) {
135                         return distances[i].instance.getVector()
136                                 .get(classIndex);
137                     }
138                     double weight = weighted ? (1 / distances[i].distance)
139                             : 1.0;
140                     sumWeights += weight;
141                     sum += weight
142                             * (Double) (distances[i].instance.getVector().get(
143                                     classIndex).getValue());
144                 }
145                 return new NumericValue(sum / sumWeights);
146             } else {
147                 // 2nd case: nominal class value
148 
149                 // a map for storing frequencies/weights
150                 Map<Value, Double> frequencies = new HashMap<Value, Double>();
151                 for (int i = 0; i < distances.length; ++i) {
152                     if (distances[i].distance == 0.0) {
153                         return distances[i].instance.getVector()
154                                 .get(classIndex);
155                     }
156                     Value currentClass = distances[i].instance.getVector().get(
157                             classIndex);
158                     Double oldValue = frequencies.get(currentClass);
159                     double weight = weighted ? (1 / distances[i].distance)
160                             : 1.0; // set weight or frequency
161                     if (Double.isNaN(weight)) {
162                         weight = 1.0;
163                     }
164                     if (oldValue == null) {
165                         frequencies.put(currentClass, weight);
166                     } else {
167                         frequencies.put(currentClass, oldValue + weight);
168                     }
169                 }
170                 // traverse the map to find the most frequent class value
171                 double maxFrequency = Double.MIN_VALUE;
172                 Value result = null;
173                 for (Entry<Value, Double> entry : frequencies.entrySet()) {
174                     Value currentClass = entry.getKey();
175                     Double currentFrequency = entry.getValue();
176                     if (currentFrequency > maxFrequency) {
177                         maxFrequency = currentFrequency;
178                         result = currentClass;
179                     }
180                 }
181                 return result;
182             }
183         }

232     /*
233      * fill the queue with distances from the query instance to its K nearest
234      * neighbors
235      */
236     private void calculateDistances(Instance query) {
237 
238         double result = 0;
239         List<Value> queryVector = query.getVector();
240         int vectorSize = originalTrainSet.getAttributes().size();
241         // for optimization
242         queryVectorArray = new Value[queryVector.size()];
243         for (int i = 0; i < queryVectorArray.length; i++) {
244             queryVectorArray[i] = queryVector.get(i);
245         }
246 
247         for (Instance currentInstance : trainSet) {
248 
249             List<Value> currentVector = currentInstance.getVector();
250             result = 0;
251             for (int i = 0; i < vectorSize; ++i) {
252                 if (i == classIndex) {
253                     // skip distance beteween class values
254                     continue;
255                 }
256                 Value currentValue = currentVector.get(i);
257                 result += distanceBetweenValues(queryVectorArray[i],
258                         currentValue, attrTypes[i]);
259             }
260 
261             result = Math.sqrt(result); // distance is Euclidean
262             distanceQueue.push(result, currentInstance);
263         }
264     }

266     /* returns the distance between two values, given their type */
267     private double distanceBetweenValues(Value queryValue, Value currentValue,
268             ValueType attrType) {
269         // 1st case: nominal values
270         if (attrType == ValueType.Nominal) {
271 
272             if (queryValue.getType() != ValueType.Unknown
273                     && currentValue.getType() != ValueType.Unknown) {
274                 if (!queryValue.equals(currentValue)) {
275                     return 1;
276                 }
277             } else if (queryValue.getType() == ValueType.Unknown
278                     && currentValue.getType() == ValueType.Unknown) {
279                 return 0;
280             } else {
281                 return 1;
282             }
283         } else if (attrType == ValueType.Numeric) {
284             // 2nd case: numeric values
285             if (queryValue.getType() != ValueType.Unknown
286                     && currentValue.getType() != ValueType.Unknown) {
287                 double dst = (Double) queryValue.getValue()
288                         - (Double) currentValue.getValue();
289                 return dst * dst;
290             } else if ((queryValue.getType() == ValueType.Unknown 
291                     && currentValue.getType() == ValueType.Unknown)) {
292                 return 0;
293             } else {
294                 return 1;
295             }
296         }
297         return 0;
298     }

300     @Override
301     public void buildClassifier(Dataset instances) {
302 
303         originalTrainSet = instances;
304         classIndex = instances.getClassIndex();
305         trainSet = new ArrayList<Instance>(instances.getElements().size());
306         List<Instance> givenInstances = instances.getElements();
307         Iterator<Instance> it = givenInstances.iterator();
308         while (it.hasNext()) { // prune instances with missing class value
309             Instance current = it.next();
310             if (current.getVector().get(classIndex).getType() 
311                     != ValueType.Unknown) {
312                 trainSet.add(current);
313             }
314         }
315         // set attribute types array
316         attrTypes = new ValueType[instances.getAttributesCount()];
317         for (int i = 0; i < attrTypes.length; i++) {
318             attrTypes[i] = instances.getAttributes().get(i).getType();
319         }
320 
321         if (k > trainSet.size()) {
322             setK(trainSet.size());
323             LOG.error("KNN: k >= all neighbors! "
324                     + "Setting k to number of neighbors.");
325         }
326     }

328     @Override
329     public Value classifyInstance(Instance instance) {
330         distanceQueue.clear();
331         calculateDistances(instance);
332         return distanceQueue.voteForClass();

348     private void setParameters(String paramName, String paramValue) {
349         if (paramName.equalsIgnoreCase("k")) {
350             int kValue;
351             try {
352                 kValue = Integer.parseInt(paramValue);
353             } catch (NumberFormatException ex) {
354                 kValue = DEFAULT_K;
355                 throw new RuntimeException("\"k\" must be an integer.");
356             }
357             setK(kValue);
358         } else if (paramName.equalsIgnoreCase("weighted")) {
359             if (paramValue.equalsIgnoreCase("true")) {
360                 setWeighted(true);
361             } else if (paramValue.equalsIgnoreCase("false")) {
362                 setWeighted(false);
363             } else {
364                 throw new RuntimeException("\"weighted\" must be boolean.");
365             }
366         } else {
367             throw new IllegalArgumentException(
368                     "KNN: unknown parameter (valid: K; Weighted)");
369         }
370     }
