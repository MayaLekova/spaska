23      /**
24       * Distribute instances according to the condition they satisfy.
25       * 
26       * @param instances
27       *            the instances to distribute
28       * @param conditions
29       *            the conditions on which to distribute the instances
30       * @param datasetService
31       *            the data service for these instances
32       * @return the distribution of the instances according to the given
33       *         parameters
34       */
35      public static List<List<Instance>> distribute(List<Instance> instances,
36              List<Condition> conditions, DatasetService datasetService) {
37          List<List<Instance>> result = new ArrayList<List<Instance>>();
38          List<Instance> unknown = new ArrayList<Instance>();
39          for (int i = 0; i < conditions.size(); i++) {
40              result.add(new ArrayList<Instance>());
41          }
42          if (conditions.isEmpty()) {
43              return result;
44          }
45          // all conditions test the same attribute
46          Condition first = conditions.get(0);
47          int attributeIndex = datasetService.getAttributeIndex(first
48                  .getAttribute());
49          int totalKnown = 0;
50          // distribute instances to relevant conditions
51          for (Instance instance : instances) {
52              int listCounter = 0;
53              Value current = instance.getVector().get(attributeIndex);
54              for (Condition c : conditions) {
55                  if (current.getType() == ValueType.Unknown) {
56                      unknown.add(instance);
57                      break;
58                  }
59                  if (c.ifTrue(current)) {
60                      result.get(listCounter).add(instance);
61                      totalKnown++;
62                      break;
63                  }
64                  listCounter++;
65              }
66          }
67          // set portion of instances reaching a condition
68          int listCounter = 0;
69          for (Condition c : conditions) {
70              c.setReach(((double) result.get(listCounter).size()) / totalKnown);
71              listCounter++;
72          }
73          // handle unknown
74          for (Instance instance : unknown) {
75              int counter = 0;
76              for (Condition c : conditions) {
77                  Instance copy = (Instance) instance.clone();
78                  copy.setWeight(copy.getWeight() * c.getReach());
79                  result.get(counter).add(copy);
80                  counter++;
81              }
82          }
83          return result;
84      }

86      /**
87       * Get a service for the best numeric attribute to split on.
88       * 
89       * @param list
90       *            the list to split
91       * @param classesEntropy
92       *            the classes entropy
93       * @param datasetService
94       *            the data service for the instances
95       * @return the service for the best numeric attribute to split on
96       */
97      public static ContinuousValueService getBestNumeric(List<Instance> list,
98              double classesEntropy, DatasetService datasetService) {
99          int[] numericIndices = datasetService.getNumericIndices();
100         ContinuousValueService best = ContinuousValueService
101                 .createEmptyService();
102         ContinuousValueService currentService;
103         double max = 0, currentRatio;
104         for (int i = 0; i < numericIndices.length; i++) {
105             currentService = new ContinuousValueService(datasetService, list,
106                     numericIndices[i], classesEntropy);
107             currentRatio = currentService.getGainRatio();
108             if (max < currentRatio) {
109                 max = currentRatio;
110                 best = currentService;
111             }
112         }
113         return best;
114     }

117      * Get conditions (nodes in the tree) for a nominal attribute.
118      * 
119      * @param attributeIndex
120      *            the attribute index
121      * @param majorityClass
122      *            the majority class
123      * @param datasetService
124      *            the data service to use as an utility
125      * @return conditions (nodes in the tree) for a nominal attribute
126      * 
127      */
128     public static List<Condition> getNominalConditions(int attributeIndex,
129             Value majorityClass, DatasetService datasetService) {
130         List<Condition> children = new ArrayList<Condition>();
131         Attribute a = datasetService.getAttribute(attributeIndex);
132         for (Value val : datasetService.getAttributeDomain(attributeIndex)) {
133             children.add(new Condition(a, val, Sign.EQ, majorityClass));
134         }
135         return children;
136     }

154         List<Condition> children = new ArrayList<Condition>();
155         Attribute a = datasetService.getAttribute(attributeIndex);
156         Value doubleValue = new NumericValue(splitValue);
