45      // get majority class of instances
46      private Value majorityVote(List<Instance> instances) {
47          int classIndex = datasetService.classIndex();
48          int[] freq = new int[datasetService.numberOfClasses()];
49          Value[] match = new NominalValue[datasetService.numberOfClasses()];
50          for (Instance instance : instances) {
51              int classInt = datasetService.intValue(classIndex,
52                      datasetService.getClass(instance));
53              match[classInt] = datasetService.getClass(instance);
54              freq[classInt]++;
55          }
56          int max = 0, maxIndex = 0;
57          for (int i = 0; i < freq.length; i++) {
58              if (max < freq[i]) {
59                  max = freq[i];
60                  maxIndex = i;
61              }
62          }
63          return match[maxIndex];
64      }

66      // get index of best nominal attribute (according to information theory)
67      private int getBestNominal(NominalInfoService service) {
68          int[] nominalIndices = datasetService.getNominalIndices();
69          double max = 0, currentRatio;
70          int index = -1;
71          for (int i = 0; i < nominalIndices.length; i++) {
72              currentRatio = service.getGainRatio(nominalIndices[i]);
73              if (max < currentRatio) {
74                  max = currentRatio;
75                  index = nominalIndices[i];
76              }
77          }
78          return index;
79      }

81      // get children of the current node
82      private List<Condition> getConditions(List<Instance> list) {
83          List<Condition> children = new ArrayList<Condition>();
84          boolean[] used = new boolean[datasetService.numberOfAttributes()];
85          used[datasetService.classIndex()] = true;
86          NominalInfoService infoService = new NominalInfoService(datasetService,
87                  list, used);
88          ContinuousValueService continuousService = Trees.getBestNumeric(list,
89                  infoService.getClassesEntropy(), datasetService);
90          if (infoService.isEmpty() && continuousService.isEmpty()) {
91              return children;
92          }
93          Value majorityClass = infoService.getSiblingsMajorityClass();
94          int nominalIndex = getBestNominal(infoService);
95          if (nominalIndex < 0 && continuousService.isEmpty()) {
96              return children;
97          }
98          if (continuousService.isEmpty()
99                  || infoService.getGainRatio(nominalIndex) >= continuousService
100                         .getGainRatio()) {
101             children = Trees.getNominalConditions(nominalIndex, majorityClass,
102                     datasetService);
103         } else {
104             children = Trees.getNumericConditions(
105                     continuousService.getAttributeIndex(),
106                     continuousService.getSplitValue(), majorityClass,
107                     datasetService);
108         }
109         return children;
110     }

112     private void buildRules(Dataset data) {
113         List<Condition> list = getConditions(data.getElements());
114         if (list.isEmpty()) {
115             ZeroR z = new ZeroR();
116             z.buildClassifier(data);
117             Attribute a = datasetService.getAttribute(0);
118             Value threshold = data.getElements().get(0).getVector().get(0);
119             Condition eqCondition = new Condition(a, threshold, Sign.EQ);
120             eqCondition.setMajorityClass(z.getCommonValue());
121             eqCondition.setEffect(z.getCommonValue());
122             Condition ineqCondition = new Condition(a, threshold, Sign.NEQ);
123             ineqCondition.setEffect(z.getCommonValue());
124             rules.add(eqCondition);
125             rules.add(ineqCondition);
126         } else {
127             List<List<Instance>> distribution = Trees.distribute(
128                     data.getElements(), list, datasetService);
129             int listCounter = 0;
130             for (Condition c : list) {
131                 List<Instance> currentList = distribution.get(listCounter);
132                 c.setEffect(majorityVote(currentList));
133                 rules.add(c);
134                 listCounter++;
135             }
136         }
137         // System.out.println(this.toString());
138     }

148     @Override
149     public Value classifyInstance(Instance instance) {
150         for (Condition c : rules) {
151             int aIndex = datasetService.getAttributeIndex(c.getAttribute());
152             Value aValue = instance.getVector().get(aIndex);
153             // System.out.println(aValue.getType() == ValueType.Unknown);
154             // System.out.println(c.ifTrue(aValue));
155             // System.out.println(aValue + " " + aValue.getType());
156             if (aValue.getType() == ValueType.Unknown) {
157                 return c.getMajorityClass();
158             }
159             if (c.ifTrue(aValue)) {
160                 return c.getEffect();
161             }
162         }
163         return null;
164     }
