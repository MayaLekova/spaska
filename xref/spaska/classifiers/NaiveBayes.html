24  /**
25   * A simple implementation of the Naive Bayes classifier. Probabilistic
26   * classifier based on applying Bayes' theorem with strong (naive) independence
27   * assumptions. In simple terms, a naive Bayes classifier assumes that the
28   * presence (or absence) of a particular feature of a class is unrelated to the
29   * presence (or absence) of any other feature, given the class variable. Note:
30   * The algorithm uses density function to calculate the posterior probability of
31   * every class based on the instances' features
32   * 
33   * @see http://en.wikipedia.org/wiki/Naive_Bayes_classifier
34   * @author Gergana Dzhumerkova
35   * 
36   */
37  public class NaiveBayes implements IClassifier {

88      @Override
89      public void buildClassifier(Dataset instances) {
90          dataService = new DatasetService(instances);
91          int numberOfClasses = dataService.numberOfClasses();
92          values = new MeanVariance[numberOfClasses][dataService
93                  .numberOfAttributes() - 1];
94  
95          int classAttributeIndex = dataService.classIndex();
96          for (Attribute a : instances.getAttributes()) {
97              int attributeIndex = dataService.getAttributeIndex(a);
98              if (attributeIndex == dataService.classIndex()) {
99                  continue;
100             }
101             for (Instance instance : instances.getElements()) {
102                 Value clazz = dataService.getClass(instance);
103                 int classIndex = dataService.intValue(classAttributeIndex,
104                         clazz);
105                 MeanVariance meanVar = getMeanVar(classIndex, attributeIndex);
106 
107                 Value value = instance.getVector().get(attributeIndex);
108                 double attributeValue = getDoubleValue(attributeIndex, value);
109                 meanVar.addValue(attributeValue);
110             }
111         }
112     }

114     private MeanVariance getMeanVar(int classIndex, int attributeIndex) {
115         MeanVariance meanVar = values[classIndex][attributeIndex];
116         if (meanVar == null) {
117             values[classIndex][attributeIndex] = new MeanVariance();
118         }
119         return values[classIndex][attributeIndex];
120     }
121 
122     private double getDoubleValue(int attributeIndex, Value value) {
123         if (ValueType.Numeric.equals(value.getType())) {
124             return ((NumericValue) value).getValue();
125         } else if (ValueType.Nominal.equals(value.getType())) {
126             return dataService.intValue(attributeIndex, value) + 1;
127         } else if (ValueType.Unknown.equals(value.getType())) {
128             return 1.0D;
129         } else {
130             throw new UnsupportedOperationException("Invalid value type!");
131         }
132     }

134     static double normalDensityF(double mean, double variance, double x) {
135         if (variance == 0.0D) {
136             return 1.0D;
137         }
138         double exp = exp((-(x - mean) * (x - mean)) / (2 * variance));
139         double d = 1 / (sqrt(2 * PI * variance));
140         return d * exp;
141     }

143     @Override
144     public Value classifyInstance(Instance instance) {
145 
146         double classProbability = 1.0D / values.length;
147         int maxClassIndex = -1;
148         double maxPosteriorNumerator = -Double.MAX_VALUE;
149 
150         List<Value> vector = instance.getVector();
151         for (int classIndex = 0; classIndex < values.length; classIndex++) {
152             double posteriorNumerator = classProbability;
153             if (values[classIndex][0] == null) {
154                 continue;
155             }
156             for (int attributeIndex = 0; attributeIndex < values[0].length; attributeIndex++) {
157                 double mean = values[classIndex][attributeIndex].getMean();
158                 double variance = values[classIndex][attributeIndex]
159                         .getVariance();
160                 double value = getDoubleValue(attributeIndex,
161                         vector.get(attributeIndex));
162                 double density = normalDensityF(mean, variance, value);
163                 posteriorNumerator *= density;
164             }
165             if (Double.isNaN(posteriorNumerator)
166                     || Double.isInfinite(posteriorNumerator)) {
167                 LOG.warn("Ignoring class value as posterior probability is non a valid double value!");
168             }
169             if (posteriorNumerator > maxPosteriorNumerator) {
170                 maxPosteriorNumerator = posteriorNumerator;
171                 maxClassIndex = classIndex;
172             }
173         }
174         Value value = dataService.getValueFromInt(dataService.classIndex(),
175                 maxClassIndex);
176         return value;
177     }
