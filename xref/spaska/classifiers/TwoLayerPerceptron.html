103     private void initNets() {
104         firstNodes = new double[numAttributes];
105         innerNodes = new double[hiddenNodes];
106         classNodes = new double[numClasses];
107         delta = new double[numClasses];
108         bNet = new double[numAttributes][hiddenNodes];
109         eNet = new double[hiddenNodes][numClasses];
110         for (int i = 0; i < bNet.length; i++) {
111             for (int j = 0; j < bNet[0].length; j++) {
112                 bNet[i][j] = randomWeight();
113             }
114         }
115         for (int i = 0; i < eNet.length; i++) {
116             for (int j = 0; j < eNet[0].length; j++) {
117                 eNet[i][j] = randomWeight();
118             }
119         }
120     }

200     private double randomWeight() {
201         int sign = randomObject.nextInt(2) > 0 ? 1 : -1;
202         return randomObject.nextDouble() * DEFAULT_INTERVAL_LENGTH * sign;
203     }

205     // get double representation of attribute value
206     private double getDoubleValue(Instance instance, int attributeIndex) {
207         Value attributeValue = instance.getVector().get(attributeIndex);
208         double result = 0;
209         switch (attributeValue.getType()) {
210         case Numeric:
211             result = (Double) attributeValue.getValue();
212             break;
213         case Nominal:
214             result = datasetService.intValue(attributeIndex, 
215                     attributeValue) + 1;
216             if (attributeIndex == datasetService.classIndex()) {
217                 result--;
218             }
219             break;
220         default:
221             break;
222         }
223         return result;
224     }

226     // forward pass in the network
227     private void propagateForward(Instance instance) {
228         double sum;
229         for (int i = 0; i < nonclassIndices.length; i++) {
230             firstNodes[i] = getDoubleValue(instance, nonclassIndices[i]);
231         }
232         for (int i = 0; i < hiddenNodes; i++) {
233             innerNodes[i] = 0;
234             sum = 0;
235             for (int j = 0; j < nonclassIndices.length; j++) {
236                 sum += firstNodes[j] * bNet[j][i];
237             }
238             innerNodes[i] = aFunc(sum);
239         }
240         for (int i = 0; i < numClasses; i++) {
241             classNodes[i] = 0;
242             sum = 0;
243             for (int j = 0; j < hiddenNodes; j++) {
244                 sum += innerNodes[j] * eNet[j][i];
245             }
246             classNodes[i] = aFunc(sum);
247         }
248     }
249 
250     private void propagateBackward(Instance instance) {
251         double t = INITIAL_VALUE;
252         for (int j = 0; j < numClasses; j++) {
253             t = target(j, getDoubleValue(instance, 
254                     datasetService.classIndex()));
255             delta[j] = derivative(classNodes[j]) * (t - classNodes[j]);
256             for (int i = 0; i < hiddenNodes; i++) {
257                 eNet[i][j] += delta[j] * innerNodes[i] * learningRate;
258             }
259         }
260         double deltaJ, sum;
261         for (int j = 0; j < hiddenNodes; j++) {
262             deltaJ = derivative(innerNodes[j]);
263             sum = 0;
264             for (int k = 0; k < numClasses; k++) {
265                 sum += delta[k] * eNet[j][k];
266             }
267             deltaJ *= sum;
268             for (int i = 0; i < nonclassIndices.length; i++) {
269                 bNet[i][j] += deltaJ * firstNodes[i] * learningRate;
270             }
271         }
272     }

275     private double sigmoid(double y) {
276         return 1 / (1 + Math.exp(-y));
277     }

298     // training with a preset number of epochs
299     private void train() {
300         for (int i = 0; i < numEpochs; i++) {
301             for (int j = 0; j < data.size(); j++) {
302                 propagateForward(data.get(j));
303                 propagateBackward(data.get(j));
304             }
305         }
306     }

308     // max number in a double array
309     private int max(double[] a) {
310         int index = 0;
311         double max = Integer.MIN_VALUE;
312         for (int i = 0; i < a.length; i++) {
313             if (max < a[i]) {
314                 max = a[i];
315                 index = i;
316             }
317         }
318         return index;
319     }

322     @Override
323     public void buildClassifier(Dataset instances) {
324 
325         if (instances == null || instances.getElements().size() <= 0) {
326             throw new NullPointerException(
327                     "Two Layer Perceptron : Instances cannot be null");
328         }
329 
330         this.datasetService = new DatasetService(instances);
331         this.data = instances.getElements();
332         this.numAttributes = datasetService.numberOfAttributes();
333         this.numClasses = datasetService.numberOfClasses();
334         if (calculateDefault) {
335             this.hiddenNodes = (numAttributes + numClasses) / 2;
336         }
337         this.nonclassIndices = new int[numAttributes - 1];
338         int j = 0;
339         for (int i = 0; i < numAttributes; i++) {
340             if (i != datasetService.classIndex()) {
341                 nonclassIndices[j++] = i;
342             }
343         }
344         initNets();
345         train();
346     }

350         propagateForward(instance);
351         int intValue = max(classNodes);

368     private void setParameter(String paramName, String paramValue) {
369         if (paramName.equalsIgnoreCase(EPOCHS_NAME)) {
370             try {
371                 setNumberOfEpochs(Integer.parseInt(paramValue));
372             } catch (Exception e) {
373                 setNumberOfEpochs(DEFAULT_NUMBER_OF_ITERATIONS);
374             }
375             return;
376         }
377         if (paramName.equalsIgnoreCase(LEARNING_RATE_NAME)) {
378             try {
379                 setLearningRate(Double.parseDouble(paramValue));
380             } catch (Exception e) {
381                 setLearningRate(DEFAULT_LEARNING_RATE);
382             }
383             return;
384         }
385         if (paramName.equalsIgnoreCase(HIDDEN_NODES_NAME)) {
386             if (paramValue.equalsIgnoreCase(HIDDEN_NODES_DEFAULT)) {
387                 calculateDefault = true;
388             } else {
389                 try {
390                     setHiddenNodes(Integer.parseInt(paramValue));
391                     calculateDefault = false;
392                 } catch (Exception e) {
393                     calculateDefault = true;
394                 }
395             }
396             return;
397         }
