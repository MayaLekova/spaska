101 	private static ClassifierStatistics test(IClassifier classifier,
102 			Dataset testingSet) {
103 		Set<Value> classDomain = testingSet.getAllClassNamesSet();
104 		String[] classNames = new String[classDomain.size()];
105 		int i = 0;
106 		for (Value v : classDomain) {
107 			classNames[i++] = v.getValue().toString();
108 		}
109 		ClassifierStatistics cs = new ClassifierStatistics(ValueType.Nominal);
110 		for (Instance inst : testingSet.getElements()) {
111 			cs.addNominalInfo(testingSet.getClassName(inst), classifier
112 					.classifyInstance(inst).getValue().toString());
113 		}
114 		return cs;
115 	}
